📦 SQLite Table Specifications
👤
users
User profiles, subscription tiers, daily quota limits & streak records.
8 Columns 1 Index
Column Name Data Type Key & Constraints Default / Value Description & Business Purpose
id TEXT PK - UUID v4 identifier for the local user profile.
email TEXT UNIQUE - User email address for cloud sync or authentication.
subscription_tier TEXT DEFAULT 'FREE' Subscription level: 'FREE', 'PRO_CAPY', or 'VIP'.
daily_scans_left INTEGER DEFAULT 3 Quota control for AI vision photo analysis per day.
daily_chat_left INTEGER DEFAULT 10 Quota control for Capybara AI Tutor chat messages per day.
last_scan_date TEXT - NULL ISO8601 string (YYYY-MM-DD) tracking last quota reset.
total_yuzu_coins INTEGER DEFAULT 0 Gamification currency earned from gameplay to buy Capy items.
streak_count INTEGER DEFAULT 0 Consecutive active learning days (Duolingo style streak).
google_id TEXT NULLABLE NULL Google OAuth ID for social authentication & Cloud Sync.
avatar_url TEXT NULLABLE NULL Google profile avatar image URL or custom mascot path.
⚡ OPTIMIZED INDEXES
Accelerates user auth, email lookup & Google SSO query
CREATE INDEX idx_users_email ON users(email); CREATE INDEX idx_users_google ON users(google_id);
🖼️
photos
Captured AI vocabulary photos, Skia rendered image URIs & path map status.
8 Columns 1 Index 1 FK
Column Name Data Type Key & Constraints Default / Value Description & Business Purpose
id TEXT PK - UUID v4 photo entry identifier.
user_id TEXT FK - Foreign key referencing users(id).
album_name TEXT NULLABLE NULL Optional album category name (e.g. 'Coffee Shop', 'Kitchen').
original_uri TEXT NOT NULL - Local file URI of raw camera image.
annotated_uri TEXT NOT NULL - Local file URI of RN ViewShot / Skia exported image.
template_id TEXT DEFAULT 'capy_pastel' Skia overlay theme template ('capy_pastel', 'neon_yuzu').
is_completed INTEGER DEFAULT 0 Boolean flag (0/1) for Path Node level completion.
created_at TEXT NOT NULL - ISO8601 timestamp of photo capture.
⚡ OPTIMIZED INDEX
Optimized for Photo Path Map Nodes & Timeline chronological sorting
CREATE INDEX idx_photos_user_created ON photos(user_id, created_at ASC);
📌
photo_notes
Detected target words, relative percent coordinates & Skia canvas overlay pins.
8 Columns 1 Index CASCADE
Column Name Data Type Key & Constraints Default / Value Description & Business Purpose
id TEXT PK - UUID v4 note pin identifier.
photo_id TEXT FK - Foreign key photos(id) ON DELETE CASCADE.
word TEXT NOT NULL - Target English vocabulary word detected by AI Vision.
phonetic TEXT NULLABLE NULL IPA phonetic notation (e.g. /kæpɪˈbɑːrə/).
vietnamese_meaning TEXT NOT NULL - Vietnamese translation for display on Skia box & game.
order_index INTEGER NOT NULL - Sequential number (1, 2, 3...) for Path Node Numbered Game.
pos_x_percent REAL NOT NULL - Horizontal relative percentage coordinate (0.0 to 100.0%).
pos_y_percent REAL NOT NULL - Vertical relative percentage coordinate (0.0 to 100.0%).
⚡ OPTIMIZED INDEX
Optimized for Skia Overlay rendering & Numbered Game sequence loading
CREATE INDEX idx_notes_photo_order ON photo_notes(photo_id, order_index ASC);
📖
vocabularies
Master vocabulary bank store with AI context sentences and phonetic details.
8 Columns 1 Index 2 FKs
Column Name Data Type Key & Constraints Default / Value Description & Business Purpose
id TEXT PK - UUID v4 vocabulary master ID.
photo_id TEXT FK - Foreign key referencing origin photos(id).
user_id TEXT FK - Foreign key referencing owner users(id).
word TEXT NOT NULL - English vocabulary word string.
phonetic TEXT NULLABLE NULL Phonetic pronunciation string.
vietnamese_meaning TEXT NOT NULL - Vietnamese translation text.
context_sentence_en TEXT NULLABLE NULL AI generated example sentence in English.
context_sentence_vi TEXT NULLABLE NULL AI generated example sentence in Vietnamese.
⚡ OPTIMIZED INDEX
Optimized for Batch Multi-Select & Photo Vocab list filtering
CREATE INDEX idx_vocab_user_photo ON vocabularies(user_id, photo_id);
🎴
srs_cards
Spaced Repetition System (SM-2) engine calculation parameters & review schedule queue.
8 Columns ⚡ Super Index 1 UNIQUE FK
Column Name Data Type Key & Constraints Default / Value Description & Business Purpose
id TEXT PK - UUID v4 SRS record identifier.
vocab_id TEXT FK UNIQUE - 1-to-1 relationship with vocabularies(id).
user_id TEXT FK - Foreign key referencing owner users(id).
repetitions INTEGER DEFAULT 0 SM-2 count of consecutive successful reviews.
interval_days REAL DEFAULT 1.0 Current SM-2 review interval in days.
easiness_factor REAL DEFAULT 2.5 SM-2 Easiness Factor (EF) parameter (min 1.3).
next_review_at TEXT NOT NULL - ISO8601 datetime string indicating exact due time.
status TEXT DEFAULT 'NEW' Status enum: 'NEW', 'LEARNING', 'REVIEW', 'GRADUATED'.
⚡ SUPER OPTIMIZED INDEX (Partial Index)
Instant SM-2 Due Flashcard Query — Filters out graduated items for high speed review load
CREATE INDEX idx_srs_due_cards ON srs_cards(user_id, next_review_at ASC) WHERE status != 'GRADUATED';

5 Columns 1 Index 1 FK
Column Name Data Type Key & Constraints Default / Value Description & Business Purpose

📜 Complete SQLite Production DDL Migration Script
-- ====================================================================== -- CapyVocab Complete Production SQLite Database Schema DDL -- Framework: Expo SQLite (React Native Offline-First) -- ====================================================================== PRAGMA foreign_keys = ON; -- 1. Users Table CREATE TABLE IF NOT EXISTS users ( id TEXT PRIMARY KEY, email TEXT UNIQUE, google_id TEXT, avatar_url TEXT, subscription_tier TEXT DEFAULT 'FREE', daily_scans_left INTEGER DEFAULT 3, daily_chat_left INTEGER DEFAULT 10, last_scan_date TEXT, total_yuzu_coins INTEGER DEFAULT 0, streak_count INTEGER DEFAULT 0 ); CREATE INDEX IF NOT EXISTS idx_users_email ON users(email); CREATE INDEX IF NOT EXISTS idx_users_google ON users(google_id); -- 2. Photos Table CREATE TABLE IF NOT EXISTS photos ( id TEXT PRIMARY KEY, user_id TEXT NOT NULL, album_name TEXT, original_uri TEXT NOT NULL, annotated_uri TEXT NOT NULL, template_id TEXT DEFAULT 'capy_pastel', is_completed INTEGER DEFAULT 0, is_public INTEGER DEFAULT 0, created_at TEXT NOT NULL, FOREIGN KEY (user_id) REFERENCES users(id) ON DELETE CASCADE ); CREATE INDEX IF NOT EXISTS idx_photos_user_created ON photos(user_id, created_at ASC); CREATE INDEX IF NOT EXISTS idx_photos_public ON photos(user_id, is_public) WHERE is_public = 1; -- 3. Photo Notes Table CREATE TABLE IF NOT EXISTS photo_notes ( id TEXT PRIMARY KEY, photo_id TEXT NOT NULL, word TEXT NOT NULL, phonetic TEXT, vietnamese_meaning TEXT NOT NULL, order_index INTEGER NOT NULL, pos_x_percent REAL NOT NULL, pos_y_percent REAL NOT NULL, FOREIGN KEY (photo_id) REFERENCES photos(id) ON DELETE CASCADE ); CREATE INDEX IF NOT EXISTS idx_notes_photo_order ON photo_notes(photo_id, order_index ASC); -- 4. Vocabularies Master Bank CREATE TABLE IF NOT EXISTS vocabularies ( id TEXT PRIMARY KEY, photo_id TEXT, user_id TEXT NOT NULL, word TEXT NOT NULL, phonetic TEXT, vietnamese_meaning TEXT NOT NULL, context_sentence_en TEXT, context_sentence_vi TEXT, FOREIGN KEY (photo_id) REFERENCES photos(id) ON DELETE SET NULL, FOREIGN KEY (user_id) REFERENCES users(id) ON DELETE CASCADE ); CREATE INDEX IF NOT EXISTS idx_vocab_user_photo ON vocabularies(user_id, photo_id); -- 5. Spaced Repetition SRS Cards (SM-2 Algorithm) CREATE TABLE IF NOT EXISTS srs_cards ( id TEXT PRIMARY KEY, vocab_id TEXT UNIQUE NOT NULL, user_id TEXT NOT NULL, repetitions INTEGER DEFAULT 0, interval_days REAL DEFAULT 1.0, easiness_factor REAL DEFAULT 2.5, next_review_at TEXT NOT NULL, status TEXT DEFAULT 'NEW', FOREIGN KEY (vocab_id) REFERENCES vocabularies(id) ON DELETE CASCADE, FOREIGN KEY (user_id) REFERENCES users(id) ON DELETE CASCADE ); CREATE INDEX IF NOT EXISTS idx_srs_due_cards ON srs_cards(user_id, next_review_at ASC) WHERE status != 'GRADUATED'; -- 6. Capybara Pet Inventory CREATE TABLE IF NOT EXISTS pet_inventory ( id TEXT PRIMARY KEY, user_id TEXT NOT NULL, item_id TEXT NOT NULL, category TEXT NOT NULL, is_equipped INTEGER DEFAULT 0, FOREIGN KEY (user_id) REFERENCES users(id) ON DELETE CASCADE ); CREATE INDEX IF NOT EXISTS idx_inventory_user_equip ON pet_inventory(user_id, is_equipped); -- 7. AI Capy Chat History CREATE TABLE IF NOT EXISTS chat_messages ( id TEXT PRIMARY KEY, user_id TEXT NOT NULL, vocab_id TEXT, sender TEXT NOT NULL, message TEXT NOT NULL, created_at TEXT DEFAULT CURRENT_TIMESTAMP, FOREIGN KEY (user_id) REFERENCES users(id) ON DELETE CASCADE, FOREIGN KEY (vocab_id) REFERENCES vocabularies(id) ON DELETE SET NULL ); CREATE INDEX IF NOT EXISTS idx_chat_vocab ON chat_messages(user_id, vocab_id); -- 8. Friendships Social Table CREATE TABLE IF NOT EXISTS friendships ( id TEXT PRIMARY KEY, requester_id TEXT NOT NULL, addressee_id TEXT NOT NULL, status TEXT DEFAULT 'PENDING', created_at TEXT DEFAULT CURRENT_TIMESTAMP, FOREIGN KEY (requester_id) REFERENCES users(id) ON DELETE CASCADE, FOREIGN KEY (addressee_id) REFERENCES users(id) ON DELETE CASCADE, UNIQUE(requester_id, addressee_id) ); CREATE INDEX IF NOT EXISTS idx_friendships_lookup ON friendships(requester_id, addressee_id, status); -- 9. App Notifications Table (Notification Center) CREATE TABLE IF NOT EXISTS app_notifications ( id TEXT PRIMARY KEY, user_id TEXT NOT NULL, title TEXT NOT NULL, body TEXT NOT NULL, type TEXT NOT NULL, -- 'SRS_REMINDER' | 'FRIEND_REQUEST' | 'SYSTEM_REWARD' | 'LEADERBOARD_ALERT' target_screen TEXT, target_params TEXT, is_read INTEGER DEFAULT 0, -- 0: Unread, 1: Read created_at TEXT ISO8601 NOT NULL, FOREIGN KEY (user_id) REFERENCES users(id) ON DELETE CASCADE ); CREATE INDEX IF NOT EXISTS idx_notif_unread ON app_notifications(user_id, is_read) WHERE is_read = 0; CREATE INDEX IF NOT EXISTS idx_notif_created ON app_notifications(user_id, created_at DESC); -- 10. Chat Conversations Table (P2P Friend Direct Chat) CREATE TABLE IF NOT EXISTS chat_conversations ( id TEXT PRIMARY KEY, participant1_id TEXT NOT NULL REFERENCES users(id) ON DELETE CASCADE, participant2_id TEXT NOT NULL REFERENCES users(id) ON DELETE CASCADE, last_message TEXT NULLABLE, last_message_at TEXT ISO8601, unread_count_p1 INTEGER DEFAULT 0, unread_count_p2 INTEGER DEFAULT 0, UNIQUE(participant1_id, participant2_id) ); CREATE INDEX IF NOT EXISTS idx_conv_user ON chat_conversations(participant1_id, participant2_id); -- 11. Direct Chat Messages Table (P2P History & Vocab Sharing) CREATE TABLE IF NOT EXISTS direct_chat_messages ( id TEXT PRIMARY KEY, conversation_id TEXT NOT NULL REFERENCES chat_conversations(id) ON DELETE CASCADE, sender_id TEXT NOT NULL REFERENCES users(id) ON DELETE CASCADE, receiver_id TEXT NOT NULL REFERENCES users(id) ON DELETE CASCADE, message_type TEXT DEFAULT 'TEXT', -- 'TEXT' | 'SHARED_VOCAB' | 'SHARED_PHOTO' text_content TEXT NOT NULL, attached_vocab_id TEXT NULLABLE REFERENCES vocabularies(id) ON DELETE SET NULL, is_read INTEGER DEFAULT 0, created_at TEXT ISO8601 NOT NULL ); CREATE INDEX IF NOT EXISTS idx_direct_chat_conv ON direct_chat_messages(conversation_id, created_at ASC);
